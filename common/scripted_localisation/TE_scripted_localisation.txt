# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { 
	name = TE_GET_GOVERNMENT_TEXT
	# Dissolve
	text = {	
		trigger = {			
			check_variable = { ROOT.var_TE_dissolve@var:ROOT.var_TE_target_country = 1 }
			THIS = { is_owned_by = var:ROOT.var_TE_target_country }
		}
		localization_key = TE_country_dissolve
	}
	#  Recog
	text = {	
		trigger = {			
			check_variable = { ROOT.var_TE_get_recog@var:ROOT.var_TE_target_country = 1 }
			THIS = { is_owned_by = var:ROOT.var_TE_target_country }
		}
		localization_key = TE_country_recognize
	}
	# Chnage GOv
	text = {	
		trigger = {			
			var:ROOT.var_TE_target_country = { check_variable = { var_TE_change_gov@ROOT = 1 } }
			THIS = { is_owned_by = var:ROOT.var_TE_target_country }
		}
		localization_key = TE_country_change_gov_to_us
	}
	# Puppet
	text = {	
		trigger = {			
			check_variable = { ROOT.var_TE_inst_pup@var:ROOT.var_TE_target_country = 1 }
			THIS = { is_owned_by = var:ROOT.var_TE_target_country }
		}
		localization_key = TE_country_puppet_this
	}
	text = {	
		trigger = {			
			var:ROOT.var_TE_target_country = { check_variable = { var_TE_change_gov@ROOT = 0 } }
			THIS = { is_owned_by = var:ROOT.var_TE_target_country }
		}
		localization_key = TE_country_not_change_gov_to_us
	}
	
	
}

###### SELECTED STATE
defined_text = { 
	name = TE_GET_SELECTED_STATE_TEXT
	# OWned by target and core of ROOT
	text = {
		
		trigger = {			
			FROM = {is_core_of = ROOT }
			var:ROOT.var_TE_target_country = { owns_state = PREV }
		}
		localization_key = TE_state_is_core
	}
	# Owned by target and claimed by ROOT
	text = {
		trigger = {
			var:ROOT.var_TE_target_country = { owns_state = PREV }
			FROM = {is_claimed_by = ROOT }
		}
		localization_key = TE_state_is_claimed
	}
	# Owned by target and bordered by ROOT
	text = {
		trigger = {
			var:ROOT.var_TE_target_country = { owns_state = PREV }
			any_neighbor_state = { is_owned_by = ROOT }
		}
		localization_key = TE_state_is_border
	}
	# Owned by target
	text = {
		trigger = {
			var:ROOT.var_TE_target_country = { owns_state = PREV }
		}
		localization_key = TE_state_is_owned_by_target
	}
	# Owned by ROOT and core Target
	text = {
		trigger = {
			FROM = { is_owned_by = ROOT }
			is_core_of = var:ROOT.var_TE_target_country 
		}
		localization_key = TE_state_is_core_of_target
	}
	# Owned by ROOT and claimed by Target
	text = {
		trigger = {
			is_owned_by = ROOT
			is_claimed_by = var:ROOT.var_TE_target_country 
		}
		localization_key = TE_state_is_claimed_by_target
	}
	# Owned by ROOT
	text = {
		trigger = {
			is_owned_by = ROOT
		}
		localization_key = TE_state_is_owned_by_root
	} 
	# Owned by Other
	text = {
		
		localization_key = TE_state_other
	}
	
}
