add_namespace = hf

country_event = {
	id = hf.1
	title = hf.1.t
	desc = hf.1.d
	picture = GFX_strike_breaking

	trigger = {
		check_variable = {
			var = life_counter
			value = 20
			compare = less_than
		}
		NOT = {
			has_country_flag = hflib
		}
	}

	option = {
		name = hf.1.a
		trigger = {
			OR = {
				has_government = fascism
				has_government = democratic
			}
		}
		add_dynamic_modifier = {
			modifier = hf_li_bad
		}
		add_ideas = hf_com
		set_country_flag = hflib
		add_war_support = -0.1
	}
	option = {
		name = hf.1.a
		trigger = {
			OR = {
				has_government = communism
				has_government = neutrality
			}
		}
		add_dynamic_modifier = {
			modifier = hf_li_bad
		}
		add_ideas = hf_fasci
		set_country_flag = hflib
		add_war_support = -0.1
	}
}
