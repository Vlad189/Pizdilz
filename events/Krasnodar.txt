add_namespace = kss # Специальное обозначение, которое ставится перед точкой (пример: kss.1.t или kss.1.b )

	# (тут можно написать название ивента)
country_event = {
	id = kss.1 # Уникальный идентефикатор или ID вашего ивента
	title = kss.1.t # Название ивента (нужно для локализации)
	desc = kss.1.d # Текст ивента (нужно для локализации)
	picture = GFX_report_event_soviet_tanks # Картинка вашего ивента
	
	fire_only_once = yes # Означает, что этот ивен произойдет лишь один раз за игру
	
	trigger = { # Условия, при которых ивент произойдет
		original_tag = KSS # Событие происходит только в стране с тегом SS (тег - индивидуальный идентефикатор страны)
		POV = { # OLG - это область определения (см Урок 1)
			owns_state = 245 # Проверяет, контролирует ли страна (в данном случае страна с тегом OLG) провинцию с id 245
		}
		POV = { # OLG - это область определения (см Урок 1)
			owns_state = 218 # Проверяет, контролирует ли страна (в данном случае страна с тегом OLG) провинцию с id 218
		}
	}
	
	mean_time_to_happen = { # Среднее кол-во времени в течении которого произойдет ивент
		days = 1
	}
	
	option = { # Опция в ивенте
		name = kss.1.a # Текст в опции (нужно для локализации)
		ai_chance = { factor = 80 } # Шанс, при котором ИИ выберет данную опцию
		trigger = { # Условия, при которых опция будет доступна
			
		}
		POV = { country_event = { days = 1 id = kss.2 } } # Это команда (см Урок 1) означает, что чеез один день произойдет событие с id kss.2
	}
	
	option = { # Опция в ивенте
		name = kss.1.b # Текст в опции (нужно для локализации)
		ai_chance = { factor = 20 } # Шанс, при котором ИИ выберет данную опцию
		trigger = { # Условия, при которых опция будет доступна
		POV = { country_event = { days = 1 id = kss.3 } } # Это команда (см Урок 1) означает, что чеез один день произойдет событие с id kss.3
		}
		# При выборе данной опции нечего не произойдёт
	}
}

country_event = {
	id = kss.2
	title = kss.2.t
	desc = kss.2.d
	picture = GFX_report_event_soviet_tanks
	
	trigger = {
		original_tag = POV
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { # Ответный ультиматум
		name = kss.2.a
		ai_chance = { factor = 85 }
		trigger = {
			OR = {
				WAR = {
					exists = yes
					is_puppet = yes
				}
				WAR = {
					exists = no
				}
			}
		}
	      declare_war_on = {target = KSS type = take_core_state}
		hidden_effect = {
			
		}
	}
	
	option = { # Принятие требований
		name = kss.2.b
		ai_chance = { factor = 15 }
		trigger = {
			
		}
		POV = {transfer_state = 245}
		POV = {transfer_state = 218}
                245 = {remove_core_of = KSS}
		218 = {remove_core_of = KSS}
		245 = {add_core_of = POV}
		218 = {add_core_of = POV}
		hidden_effect = {
			245 = {set_demilitarized_zone = no}
			218 = {set_demilitarized_zone = no}
		}
	}
}
