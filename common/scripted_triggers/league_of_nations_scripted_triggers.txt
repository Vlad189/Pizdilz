## set_temp_variable = { reform_id = 1 }
## can_complete_league_of_nations_reform = yes
can_complete_league_of_nations_reform = {
	var:global.league_of_nations_leader = {
		is_no_league_of_nations_reform_decision_active = yes
	}
	if = {
		limit = {
			is_subject = yes
		}
		is_subject = no
	}
	if = {
		limit = {
			check_variable = { reform_id = 1 }
		}
		set_temp_variable = { num_votings = 1 }
		has_num_of_successful_league_of_nations_votings = yes
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 2 }
		}
		set_temp_variable = { reform_id = 1 }
		has_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 3 }
		has_not_completed_league_of_nations_reform = yes
		set_temp_variable = { num_votings = 3 }
		has_num_of_successful_league_of_nations_votings = yes
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 3 }
		}
		set_temp_variable = { reform_id = 1 }
		has_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 2 }
		has_not_completed_league_of_nations_reform = yes
		set_temp_variable = { num_votings = 3 }
		has_num_of_successful_league_of_nations_votings = yes
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 4 }
		}
		set_temp_variable = { reform_id = 2 }
		has_completed_league_of_nations_reform = yes
		date > 1939.1.1
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 5 }
		}
		set_temp_variable = { reform_id = 3 }
		has_completed_league_of_nations_reform = yes
		date > 1939.1.1
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 6 }
		}
		OR = {
			AND = {
				set_temp_variable = { reform_id = 4 }
				has_completed_league_of_nations_reform = yes
			}
			AND = {
				set_temp_variable = { reform_id = 5 }
				has_completed_league_of_nations_reform = yes
			}
		}
		date > 1941.1.1
		set_temp_variable = { num_elections = 3 }
		has_num_of_won_league_of_nations_elections = yes
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 7 }
		}
		set_temp_variable = { reform_id = 4 }
		has_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 5 }
		has_not_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 6 }
		has_completed_league_of_nations_reform = yes
		date > 1943.1.1
		has_war = no
	}
	else_if = {
		limit = {
			check_variable = { reform_id = 8 }
		}
		set_temp_variable = { reform_id = 5 }
		has_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 4 }
		has_not_completed_league_of_nations_reform = yes
		set_temp_variable = { reform_id = 6 }
		has_completed_league_of_nations_reform = yes
		date > 1943.1.1
		has_war = no
	}
}

# usage:
## voting_against_this_would_have_valid_sanction_choice = yes
voting_against_this_would_have_valid_sanction_choice = {
	OR = {
		check_variable = { league_of_nations_trade_sanctions_array^num < 10 }
		any_other_country = {
			is_league_of_nations_member = yes
			NOT = { has_wargoal_against = PREV }
		}
		is_league_of_nations_member = yes
		AND = {
			set_temp_variable = { reform_id = 2 }
			has_completed_league_of_nations_reform = yes
			set_temp_variable = { threat_value = 50 }
			has_league_of_nations_threat_level = yes
			NOT = { has_idea = league_of_nations_defense_pact_target }
		}
		AND = {
			OR = {
				has_war = yes
				any_other_country = {
					PREV = {
						OR = {
							has_wargoal_against = PREV
							is_justifying_wargoal_against = PREV
						}
					}
				}
			}
			NOT = { has_idea = league_of_nations_luxury_goods_embargo_idea }
		}
		NOT = { has_idea = league_of_nations_economic_sanctions_idea }
	}
}
