
##### Checking if peace terms have been completed
TE_every_country_finished_peace_terms = {
	custom_trigger_tooltip = {
		tooltip = TE_every_country_finished_peace_terms_tt 
		# has started treaty thingy but not sent terms yet
		NOT = {
			any_country = {
				check_variable = { THIS.var_TE_has_started_treaty@ROOT = 1 }
				check_variable = { THIS.var_TE_has_sent_treaty@ROOT = 0 }
			}
		}
		# Has accepted peace offer but has not started treaty thingy
		NOT = { 
			any_country = {
				check_variable = { THIS.var_TE_peace_offer_available@ROOT = 1 }
			}
		}
		# have to ensure that mission doesn't auto complete when sending - before being accepted/rejected
		NOT = {
			any_country = {
				check_variable = { ROOT.var_TE_has_sent_offer@THIS = 1 }
				check_variable = { THIS.var_TE_has_started_treaty@ROOT = 0 }
			}
		}
	}
}

#### CHECK IF CAN START WAR
# Independence 
TE_can_start_war_of_independence = {
	custom_trigger_tooltip = {
		tooltip = TE_can_start_war_of_independence
		check_variable = { ROOT.overlord = THIS }
		
		NOT = { has_autonomy_state = IBW_LoN_mandate }
		# Different ideology
		IF = { 
			limit = { NOT = { check_variable = { ROOT.current_party_ideology_group = THIS.current_party_ideology_group }} }
			ROOT = { 
				has_stability > 0.4 
				has_war_support > 0.3
			}
			check_variable = { ROOT.party_popularity@ruling_party > 0.59 }
			check_variable = { ROOT.autonomy_ratio > 0.54 }
		}
		# Same Ideology -> requirements = higher
		ELSE = {
			IF = {
				limit = { ROOT = { tag = MAN } }
				ROOT = { NOT = { has_completed_focus = MAN_obedience }}
			}
			ROOT = { 
				has_stability > 0.45 
				has_war_support > 0.35
			}
			check_variable = { ROOT.party_popularity@ruling_party > 0.79 }
			check_variable = { ROOT.autonomy_ratio > 0.84 }
		}
	}
}
