league_of_nations_category = {

	icon = generic_formable_nations

	picture = GFX_league_of_nations_icon

	visible = {
		NOT = { has_global_flag = league_of_nations_dissolved }
		OR = {
			NOT = { league_of_nations_exists = yes }
			AND = {
				is_league_of_nations_leader = yes
				NOT = { has_global_flag = league_of_nations_rule_reforms_disabled }
			}
		}
	}

	priority = 100
}
