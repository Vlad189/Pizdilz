				set_variable = { GER.var_TE_inst_pup_bonus@POL = 2 }
			}

			IF = {
				limit = {
					check_variable = { ROOT.var_TE_war_type@FROM = 0 }
				}
				# Danzig or War
				IF = { 
					limit = { 
						ROOT = {
							tag = GER
							has_completed_focus = GER_danzig_or_war
						}
						FROM = { tag = POL }
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 3 }
					set_variable = { FROM.var_TE_war_type@ROOT = 7 }
				}
				# South Africa 
				ELSE_IF = { 
					limit = { 
						ROOT = {
							tag = SAF
							has_completed_focus = SAF_liberate_british
						}
						FROM = { tag = ENG }
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 8 }
					set_variable = { FROM.var_TE_war_type@ROOT = 7 }
				}
				ELSE_IF = { 
					limit = { 
						ROOT = {
							tag = SAF
							has_completed_focus = SAF_liberate_portugese
						}
						FROM = { tag = POR }
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 8 }
					set_variable = { FROM.var_TE_war_type@ROOT = 7 }
				}
				ELSE_IF = { 
					limit = { 
						ROOT = {
							tag = SAF
							has_completed_focus = SAF_liberate_belgian
						}
						FROM = { tag = BEL }
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 8 }
					set_variable = { FROM.var_TE_war_type@ROOT = 7 }
				}
				
				
				# Civil War
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = civil_war  
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 4 }
				}
				# Annex 
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = annex_everything
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 3 }
				}
				# Minor Conquest War goal
				ELSE_IF = {
					limit = {
						OR = {
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = take_state
								}
							}
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = take_state_focus
								}
							}
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = take_claimed_state 
								}
							}
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = take_core_state 
								}
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 2 }
				}
				
				# Border War
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = border_war_goal 
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 1 }
				}
				
				# Topple gov
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = topple_government  
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 5 }
				}
				
				# Puppet
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = puppet_wargoal_focus   
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 5 }
				}
				# Liberation
				ELSE_IF = {
					limit = {
						ROOT = {
							has_wargoal_against = {
								target = FROM
								type = liberate_wargoal   
							}
						}
					}
					set_variable = { ROOT.var_TE_war_type@FROM = 8 }
				}
				ELSE = {
					set_variable = { ROOT.var_TE_war_type@FROM = 3 }
				}
			}
			#### DEFENDERF
			# Civil war
			IF = {
				limit = {
					check_variable = { FROM.var_TE_war_type@ROOT = 0 }
				}
				IF = {
					limit = {
						OR = {
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = civil_war  
								}
							}
							check_variable = { ROOT.var_TE_war_type@FROM = 4 }
							check_variable = { ROOT.var_TE_war_type@FROM = 6 }
						}
					}
					set_variable = { FROM.var_TE_war_type@ROOT = 4 }
				}
				ELSE_IF = {
					limit = {
						OR = {
							ROOT = {
								has_wargoal_against = {
									target = FROM
									type = border_war_goal  
								}
							}
							check_variable = { ROOT.var_TE_war_type@FROM = 1 }
						}
					}
					set_variable = { FROM.var_TE_war_type@ROOT = 1 }
				}
				ELSE = {
					set_variable = { FROM.var_TE_war_type@ROOT = 7 }
				}
			}
					
		}
