#Important to put HIGHEST first
TE_get_ptp_bonuses = {
	# Apply and remove Annex bonus
	IF = {
		limit = {
			check_variable = { ROOT.var_TE_annex_bonus@var:ROOT.var_TE_target_country >0 }
		}
		multiply_variable = { ROOT.var_TE_annex_cost = ROOT.var_TE_annex_bonus@var:ROOT.var_TE_target_country }
		round_variable = ROOT.var_TE_annex_cost 
		clear_variable = ROOT.var_TE_annex_bonus@var:ROOT.var_TE_target_country
	}
	# Apply and remove Change Government bonus
	IF = {
		limit = {
			check_variable = { ROOT.var_TE_change_gov_bonus@var:ROOT.var_TE_target_country >0 }
		}
		multiply_variable = { ROOT.var_TE_change_gov_cost = ROOT.var_TE_change_gov_bonus@var:ROOT.var_TE_target_country }
		round_variable = ROOT.var_TE_change_gov_cost 
		clear_variable = ROOT.var_TE_change_gov_bonus@var:ROOT.var_TE_target_country
	}
	
	# Apply and remove Install Puppet bonus
	IF = {
		limit = {
			check_variable = { ROOT.var_TE_inst_pup_bonus@var:ROOT.var_TE_target_country >0 }
		}
		multiply_variable = { ROOT.var_TE_inst_pup_cost = ROOT.var_TE_inst_pup_bonus@var:ROOT.var_TE_target_country }
		round_variable = ROOT.var_TE_inst_pup_cost 
		clear_variable = ROOT.var_TE_inst_pup_bonus@var:ROOT.var_TE_target_country
	}
	
	
}

TE_get_total_peace_points = {
	clear_variable = ROOT.current_no_demanded_states
	clear_variable = ROOT.current_no_demanded_states_resources
	clear_variable = ROOT.current_no_demanded_states_demil
	clear_variable = ROOT.current_no_demanded_states_dismatle
	
	IF = { 
		limit = { check_variable = { ROOT.var_TE_war_type = 1 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_bw }
	}
	ELSE_IF = { 
		limit = { check_variable = { ROOT.var_TE_war_type = 2 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_mc }
	}
	ELSE_IF = { 
		limit = {check_variable = {  ROOT.var_TE_war_type = 3 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_an }
	}
	ELSE_IF = { 
		limit = { check_variable = { ROOT.var_TE_war_type = 4 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_cw }
	}
	ELSE_IF = { 
		limit = { check_variable = { ROOT.var_TE_war_type = 5 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_otg }
	}
	ELSE_IF = { 
		limit = { check_variable = { ROOT.var_TE_war_type = 6 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_iw }
	}
	ELSE_IF = { 
		limit = { check_variable = {  ROOT.var_TE_war_type = 7 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_dw }
	}
	ELSE_IF = { 
		limit = { check_variable = {  ROOT.var_TE_war_type = 8 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_lib }
	}
	ELSE_IF = { 
		limit = { check_variable = {  ROOT.var_TE_war_type = 9 } }
		set_temp_variable = { temp_cost = global.var_TE_ptp_dis }
	}
	ELSE = { 
		set_temp_variable = { temp_cost = global.var_TE_ptp_dw }
	}
	
	# multiply with war scope
	IF = {
		limit = { check_variable = { ROOT.var_TE_war_scope = 1 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_skirmish }
	}
	ELSE_IF = {
		limit = { check_variable = { ROOT.var_TE_war_scope = 2 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_limited_war }
	}
	ELSE_IF = {
		limit = { check_variable = { ROOT.var_TE_war_scope = 3 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_major_war }
	}
	ELSE_IF = {
		limit = { check_variable = { ROOT.var_TE_war_scope = 4 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_world_war }
	}
	ELSE = {
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_limited_war }
	}
	
	# multiply with peace option
	IF = {
		limit = { check_variable = { ROOT.var_TE_peace_type = 1 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_conditional }
	}
	ELSE_IF = {
		limit = { check_variable = { ROOT.var_TE_peace_type = 2 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_capitulation }
	}
	ELSE_IF = {
		limit = { check_variable = { ROOT.var_TE_peace_type = 3 } }
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_independence }
	}
	ELSE = {
		multiply_temp_variable = { temp_cost = global.var_TE_ptp_conditional }
	}
	
	#set_variable = { ROOT.debug_1 = temp_cost }
	### adding for each controlled state before multiplying
	
	
	IF = {
		limit = { 
			NOT = { check_variable = { ROOT.var_TE_war_type = 4 } }
		}
		# Extremely important States
		every_state = {
			limit = {
				is_owned_by = var:ROOT.var_TE_target_country 
				is_controlled_by = ROOT			
				check_variable = { THIS.state_strategic_value > 249 }
			}
			
			add_to_temp_variable = { ROOT.temp_cost = 10 }
		}
		#set_variable = { ROOT.debug_ext_imp = temp_cost }
		# Very important States
		every_state = {
			limit = { 
				is_owned_by = var:ROOT.var_TE_target_country 
				is_controlled_by = ROOT
				check_variable = { THIS.state_strategic_value < 250 }
				check_variable = { THIS.state_strategic_value > 149 }
			}
			
			add_to_temp_variable = { ROOT.temp_cost = 5 }
		}
		#set_variable = { ROOT.debug_very_imp = temp_cost }
		# Important states
		every_state = {
			limit = { 
				is_owned_by = var:ROOT.var_TE_target_country 
				is_controlled_by = ROOT
				check_variable = { THIS.state_strategic_value < 150 }
				check_variable = { THIS.state_strategic_value > 99 }
			}
			
			add_to_temp_variable = { ROOT.temp_cost = 1 }
		}
		#set_variable = { ROOT.debug_imp = temp_cost }
		# Not so important states
		every_state = {
			limit = { 
				is_owned_by = var:ROOT.var_TE_target_country 
				is_controlled_by = ROOT
				check_variable = { THIS.state_strategic_value < 100 }
				check_variable = { THIS.state_strategic_value > 24 }
			}
			
			add_to_temp_variable = { ROOT.temp_cost = 0.75 }
		}
		#set_variable = { ROOT.debug_not_imp = temp_cost }
		# Unimportant states but not wasteland
		every_state = {
			limit = {
				is_owned_by = var:ROOT.var_TE_target_country 
				is_controlled_by = ROOT
				check_variable = { THIS.state_strategic_value < 25 }
				check_variable = { THIS.state_strategic_value > 1 }	
			}		
			add_to_temp_variable = { ROOT.temp_cost = 0.25 }
		}
		#set_variable = { ROOT.debug_unimp = temp_cost }
		
	}
	#set_variable = { ROOT.debug_2 = temp_cost }
	# modify war score if in faction 
	IF = {
		limit = { 
			any_country = { 
				has_war_with = var:ROOT.var_TE_target_country 
				is_in_faction_with = ROOT 
				NOT = { tag = ROOT }
			}
		}
		# if faction leader - France in the Great War 
		IF = {
			limit = { ROOT = { is_faction_leader = yes } }
			multiply_temp_variable = { temp_cost = 1 }
		}
		# not leader but major - UK/US in tGW
		ELSE_IF = {
			limit = { ROOT = { is_major = yes } }
			multiply_temp_variable = { temp_cost = 0.9 }
		}
		# Not major, but they have our cores
		ELSE_IF = {
			limit = { 
				any_state = {
					is_owned_by = var:var_TE_target_country
					OR = { is_claimed_by = ROOT is_core_of = ROOT }
				}
			}
			multiply_temp_variable = { temp_cost = 0.65 }
		}
		# Not major, but we have border
		ELSE_IF = {
			limit = { 
				var:var_TE_target_country = { 
					any_owned_state = {
						any_neighbor_state = {
							is_owned_by = ROOT
						}
					}
				}
			}
			multiply_temp_variable = { temp_cost = 0.55 }
		}
		# Not major, but we both are coastal
		ELSE_IF = {
			limit = { 
				var:var_TE_target_country = { 
					any_owned_state = {
						naval_base > 1
					}
				}
				ROOT = { 
					any_owned_state = {
						naval_base > 1
					}
				}
			}
			multiply_temp_variable = { temp_cost = 0.45 }
		}
		ELSE = {
			multiply_temp_variable = { temp_cost = 0.1 }
		}
	}
	
	#set_variable = { ROOT.debug_3 = temp_cost }
	# If holding capital
	IF = { 
		limit = {
			any_state = {
				is_owned_by = var:ROOT.var_TE_target_country
				is_capital = yes
				is_controlled_by = ROOT 
			}
		}
		multiply_temp_variable = { ROOT.temp_cost = 1.25 }
	}
	#set_variable = { ROOT.debug_4 = temp_cost }
	
	
	# Doing the Turkey option in WW2 - declaring war without even being close. 
	IF = {
		limit = {
			NOT = {
				any_state = {
					is_owned_by = var:ROOT.var_TE_target_country
					is_controlled_by = ROOT
				}
			}
		}
		multiply_temp_variable = { temp_cost = 0.67 }
	}
	
	# If other factions or powers are present
	IF = {
		limit = { 
			any_country = { 
				has_war_with = var:ROOT.var_TE_target_country 
				NOT = { is_in_faction_with = ROOT }
				NOT = { tag = ROOT }
			}
		}
		multiply_temp_variable = { temp_cost = 0.9 }
	}
	

	set_variable = { ROOT.var_TE_ptp_available = temp_cost }
	round_variable = ROOT.var_TE_ptp_available
	IF = { 
		limit = { check_variable = { ROOT.var_TE_ptp_available < 1 }}
		set_variable = { ROOT.var_TE_ptp_available = 1 }
	}
	set_variable = { ROOT.var_TE_ptp_total = ROOT.var_TE_ptp_available } 
	
	set_variable = { ROOT.var_TE_ptp_used = 0 } 
	
	# calculate costs
	TE_get_change_gov_pts = yes 
	TE_get_inst_pup_pts = yes
	TE_get_lib_conq_pts = yes
	TE_get_pup_conq_pts = yes
	TE_get_dissolve_pts = yes
	TE_get_get_recog_pts = yes
	TE_get_release_pup_pts = yes
	TE_get_trans_pup_pts = yes
	TE_get_annex_pts = yes
	TE_get_take_core_pts = yes
	TE_get_take_claimed_pts = yes
	TE_get_take_border_pts = yes
	TE_get_take_other_pts = yes
	TE_get_take_port_pts = yes
	TE_get_take_island_pts = yes
	TE_get_take_colony_pts = yes
	TE_get_set_demil_pts = yes
	TE_get_dismantle_forts_pts = yes
	TE_get_reduce_mil_pts = yes
	TE_get_reduce_naval_pts = yes
	TE_get_war_rep_pts = yes
	TE_get_demob_pts = yes
	TE_get_reduce_navy_pts = yes
	TE_get_transfer_navy_pts = yes
	TE_get_revoke_core_pts = yes
	TE_get_revoke_claim_pts = yes
	TE_get_return_core_pts = yes
	TE_get_give_to_ally_pts = yes
	
	TE_get_war_rep_pts = yes
	TE_get_oil_rights_pts = yes
	
	TE_get_limit_military_pts = yes
	
	# Check bonuses
	TE_get_ptp_bonuses = yes
}
